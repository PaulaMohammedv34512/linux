# install new kernel
uname -r

rpm -ivh kernel-4.14.131_generic-862.el7.x86_64.rpm 

cat /boot/grub2/grub.cfg |grep menuentry

# modify /etc/default/grub
vi /etc/default/grub
GRUB_DEFAULT=0

grub2-mkconfig -o /boot/grub2/grub.cfg
reboot

# load ocfs2 ocfs_dlmfs
rpm -ivh ocfs2-tools-1.8.5-4.el7.centos.x86_64.rpm
cd /lib/modules/4.14.131-generic/kernel/fs/ocfs2
modprobe ocfs2 ocfs2_dlmfs
lsmod |grep ocfs2

# ocfs2 service with systemd
o2cb_ctl -C -n prdcluster -t cluster -a name=prdcluster
systemctl enable o2cb
systemctl start o2cb
systemctl status o2cb
